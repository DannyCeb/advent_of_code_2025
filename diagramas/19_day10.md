```mermaid
graph TD
    subgraph "Lógica Principal (main)"
        A[Inicio] --> B{Leer y parsear 'day10.txt'};
        B --> C{Extraer (target, n_bits) y 'buttons' para cada línea};
        C --> D{Inicializar 'res' (resultado total) a 0};
        D --> E{Iterar sobre cada conjunto de (target, n_bits, buttons)};
        E -- Para cada conjunto --> F{Llamar a la función min_steps};
        F -- Resultado Some(steps) --> G{res += steps};
        G --> E;
        F -- Resultado None --> H{Imprimir 'Inalcanzable'};
        H --> E;
        E -- Fin de la iteración --> I{Imprimir 'res' final};
        I --> J[Fin];
    end

    subgraph "Función min_steps (Búsqueda en Anchura - BFS)"
        K[Entrada: target, n_bits, buttons] --> L{Inicializar:<br/>- 'dp' (distancias) con infinito<br/>- 'queue' con el estado inicial 0<br/>- dp[0] = 0};
        L --> M{Mientras la 'queue' no esté vacía};
        M -- Sacar 'state' de la queue --> N{¿Es 'state' el 'target' T_T?};
        N -- Sí --> O{Retornar dp[state]};
        N -- No --> P{Para cada 'button' disponible};
        P -- Aplicar botón --> Q{Calcular 'new_state' (invirtiendo bits con XOR)};
        Q --> R{¿Hemos encontrado un camino más corto a 'new_state'?<br/>(dp[new_state] > dp[state] + 1)};
        R -- Sí --> S{Actualizar distancia: dp[new_state] = dp[state] + 1<br/>Añadir 'new_state' a la 'queue'};
        S --> P;
        R -- No --> P;
        P -- Fin botones --> M;
        M -- Queue vacía --> T{Retornar None (inalcanzable)};
    end

    style J fill:#2E8B57,color:#fff
    style O fill:#2E8B57,color:#fff
```
